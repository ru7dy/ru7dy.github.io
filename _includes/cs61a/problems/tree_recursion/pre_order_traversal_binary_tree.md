    def pre_order_traversal(tree):
        """
        >>> t = Tree(5, Tree(1, None, Tree(4)), Tree(7, Tree(6), Tree(8)))
        >>> pre_order_traversal(t)
        5
        1
        4
        7
        6
        8
        """
        if tree is None:
            return
        print(tree.entry)
        pre_order_traversal(tree.left)
        pre_order_traversal(tree.right)
