    def post_order_traversal(tree):
        """
        >>> t = Tree(5, Tree(1, None, Tree(4)), Tree(7, Tree(6), Tree(8)))
        >>> post_order_traversal(t)
        4
        1
        6
        8
        7
        5
        """
        if tree is None:
            return
        post_order_traversal(tree.left)
        post_order_traversal(tree.right)
        print(tree.entry)
