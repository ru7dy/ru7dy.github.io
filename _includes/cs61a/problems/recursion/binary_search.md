    def binary_search(item, lst):
        """
        >>> l = [1, 4, 5, 8, 10, 12]
        >>> binary_search(4, l)
        True
        >>> binary_search(9, l)
        False
        """
        def helper(item, lst, low, high):
          if low >= high:
              return False
          mid = (low + high) // 2
          if item == lst[mid]:
              return True
          if item < lst[mid]:
              return helper(item, lst, low, mid - 1)
          if item > lst[mid]:
              return helper(item, lst, mid + 1, high)
        return helper(item, lst, 0, len(lst))
